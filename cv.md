```
#include <string.h> 
#include <stdio.h>
std::string abbrevName(std::string name)
{
  char ch1, ch2;
  std::string result = "";
  ch1 = std::toupper((int)name.at(0));
  int pos = 0; 
  for (int i = 0; i < name.length(); i++) {
    if (name[i] == ' ') {
      pos = i;
      break;
    }
  }
  pos += 1;
  ch2 = std::toupper((int)name[pos]);
  result = result + ch1 + char('.') + ch2;
  return result;
}
```
